    voucher_value = 60.0

    applicable_to = "base_fare"

    # Set request body
    request_body = {

    }

    # Set API headers
    headers = {
        "Authorization": f"Bearer {api_key}",
        "Content-Type": "application/json"
    }

    # Send API request
    response = requests.post(api_endpoint, json=request_body, headers=headers)

    # Check API response
    if response.status_code == 201:

        return response.json()
    else:
        print(f"Error generating voucher: {response.status_code} - {response.text}")
        return None

def lambda_handler(event, context):
    # Extract parameters from the event
    pnr = event['pnr']
    first_name = event['first_name']
    last_name = event['last_name']

    # Generate voucher
    voucher_response = generate_voucher(pnr, first_name, last_name)

    # You can now use boto3 to store the voucher details in DynamoDB or other AWS services
    # For example, store the voucher in a DynamoDB table
    if voucher_response:
        dynamodb = boto3.resource('dynamodb')
        table = dynamodb.Table('Vouchers')
        table.put_item(Item={
            'PNR': pnr,
            'FirstName': first_name,
            'LastName': last_name,
            'VoucherDetails': voucher_response
        })
        return {
            'statusCode': 200,
            'body': 'Voucher created and stored in DynamoDB'
        }
    else:
        return {
            'statusCode': 500,
            'body': 'Error generating voucher'
        }
